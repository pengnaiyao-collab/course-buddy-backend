# 核心数据表设计

## 1. 用户表（users）

### 1.1 表结构

```sql
CREATE TABLE users (
    id              BIGINT UNSIGNED     NOT NULL AUTO_INCREMENT   COMMENT '用户ID',
    username        VARCHAR(50)         NOT NULL                  COMMENT '用户名（唯一）',
    email           VARCHAR(100)        NOT NULL                  COMMENT '邮箱（唯一）',
    password        VARCHAR(255)        NOT NULL                  COMMENT '密码（BCrypt哈希）',
    full_name       VARCHAR(100)        NOT NULL DEFAULT ''       COMMENT '真实姓名',
    role            ENUM('ADMIN','TEACHER','STUDENT')
                                        NOT NULL DEFAULT 'STUDENT' COMMENT '用户角色',
    enabled         TINYINT(1)          NOT NULL DEFAULT 1        COMMENT '账号是否启用',
    created_at      DATETIME(6)         NOT NULL DEFAULT NOW(6)   COMMENT '创建时间',
    updated_at      DATETIME(6)         NOT NULL DEFAULT NOW(6)
                    ON UPDATE NOW(6)                              COMMENT '更新时间',
    deleted_at      DATETIME(6)         DEFAULT NULL              COMMENT '软删除时间',
    PRIMARY KEY (id),
    UNIQUE KEY uk_username (username),
    UNIQUE KEY uk_email (email),
    KEY idx_role (role),
    KEY idx_deleted_at (deleted_at)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci
  COMMENT='用户信息表';
```

### 1.2 字段说明

| 字段 | 类型 | 约束 | 说明 |
|------|------|------|------|
| `id` | BIGINT UNSIGNED | PK, AUTO_INCREMENT | 自增主键 |
| `username` | VARCHAR(50) | NOT NULL, UNIQUE | 登录用户名，3-50字符 |
| `email` | VARCHAR(100) | NOT NULL, UNIQUE | 邮箱，用于找回密码 |
| `password` | VARCHAR(255) | NOT NULL | BCrypt 加密后的密码 |
| `full_name` | VARCHAR(100) | NOT NULL | 显示名称 |
| `role` | ENUM | NOT NULL | 角色：ADMIN/TEACHER/STUDENT |
| `enabled` | TINYINT(1) | NOT NULL, DEFAULT 1 | 0=禁用，1=启用 |

### 1.3 关联关系

```
users ←──┐
         ├── courses.teacher_id（一对多：教师发布的课程）
         ├── course_enrollments.user_id（多对多：学生选课）
         ├── notes.user_id（一对多：用户的笔记）
         ├── questions.user_id（一对多：用户提的问题）
         ├── answers.user_id（一对多：用户的回答）
         ├── assignments.created_by（一对多：教师发布的作业）
         ├── assignment_submissions.student_id（一对多：学生提交的作业）
         ├── teams.owner_id（一对多：用户创建的团队）
         └── tokens.user_id（一对多：用户的刷新Token）
```

## 2. 课程表（courses）

### 2.1 表结构

```sql
CREATE TABLE courses (
    id              BIGINT UNSIGNED     NOT NULL AUTO_INCREMENT   COMMENT '课程ID',
    title           VARCHAR(200)        NOT NULL                  COMMENT '课程标题',
    description     TEXT                                          COMMENT '课程详情描述',
    teacher_id      BIGINT UNSIGNED     NOT NULL                  COMMENT '授课教师ID',
    price           DECIMAL(10,2)       NOT NULL DEFAULT 0.00     COMMENT '课程价格',
    category        VARCHAR(100)        NOT NULL DEFAULT ''       COMMENT '课程分类',
    status          ENUM('DRAFT','PUBLISHED','ARCHIVED')
                                        NOT NULL DEFAULT 'DRAFT'  COMMENT '课程状态',
    max_students    INT                 NOT NULL DEFAULT 100      COMMENT '最大学生数',
    cover_image_url VARCHAR(500)        DEFAULT NULL              COMMENT '封面图片URL',
    created_at      DATETIME(6)         NOT NULL DEFAULT NOW(6)   COMMENT '创建时间',
    updated_at      DATETIME(6)         NOT NULL DEFAULT NOW(6)
                    ON UPDATE NOW(6)                              COMMENT '更新时间',
    deleted_at      DATETIME(6)         DEFAULT NULL              COMMENT '软删除时间',
    PRIMARY KEY (id),
    KEY idx_teacher_id (teacher_id),
    KEY idx_status_deleted (status, deleted_at),
    KEY idx_category (category),
    CONSTRAINT fk_courses_teacher FOREIGN KEY (teacher_id)
        REFERENCES users (id) ON DELETE RESTRICT
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci
  COMMENT='课程信息表';
```

### 2.2 状态流转

```
DRAFT（草稿）
   │
   │ 教师提交审核 → content_reviews 表创建审核申请
   │
   ▼
PUBLISHED（已发布）  ← 管理员审核通过
   │
   │ 教师/管理员归档
   ▼
ARCHIVED（已归档）
```

### 2.3 选课关系表（course_enrollments）

```sql
CREATE TABLE course_enrollments (
    id              BIGINT UNSIGNED     NOT NULL AUTO_INCREMENT,
    user_id         BIGINT UNSIGNED     NOT NULL                  COMMENT '学生ID',
    course_id       BIGINT UNSIGNED     NOT NULL                  COMMENT '课程ID',
    enrolled_at     DATETIME(6)         NOT NULL DEFAULT NOW(6)   COMMENT '选课时间',
    completed_at    DATETIME(6)         DEFAULT NULL              COMMENT '完成时间',
    created_at      DATETIME(6)         NOT NULL DEFAULT NOW(6),
    updated_at      DATETIME(6)         NOT NULL DEFAULT NOW(6)
                    ON UPDATE NOW(6),
    deleted_at      DATETIME(6)         DEFAULT NULL,
    PRIMARY KEY (id),
    UNIQUE KEY uk_user_course (user_id, course_id),
    KEY idx_course_id (course_id),
    CONSTRAINT fk_enrollment_user FOREIGN KEY (user_id)
        REFERENCES users (id),
    CONSTRAINT fk_enrollment_course FOREIGN KEY (course_id)
        REFERENCES courses (id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci
  COMMENT='课程选课记录表';
```

## 3. 笔记表（notes）

### 3.1 表结构

```sql
CREATE TABLE notes (
    id              BIGINT UNSIGNED     NOT NULL AUTO_INCREMENT   COMMENT '笔记ID',
    user_id         BIGINT UNSIGNED     NOT NULL                  COMMENT '作者ID',
    course_id       BIGINT UNSIGNED     DEFAULT NULL              COMMENT '关联课程ID（可选）',
    title           VARCHAR(500)        NOT NULL                  COMMENT '笔记标题',
    content         LONGTEXT                                      COMMENT '笔记内容（Markdown）',
    is_public       TINYINT(1)          NOT NULL DEFAULT 0        COMMENT '是否公开',
    version         INT                 NOT NULL DEFAULT 0        COMMENT '乐观锁版本号',
    created_at      DATETIME(6)         NOT NULL DEFAULT NOW(6),
    updated_at      DATETIME(6)         NOT NULL DEFAULT NOW(6)
                    ON UPDATE NOW(6),
    deleted_at      DATETIME(6)         DEFAULT NULL,
    PRIMARY KEY (id),
    KEY idx_user_id (user_id),
    KEY idx_course_id (course_id),
    KEY idx_is_public (is_public),
    KEY idx_user_deleted (user_id, deleted_at),
    CONSTRAINT fk_notes_user FOREIGN KEY (user_id)
        REFERENCES users (id),
    CONSTRAINT fk_notes_course FOREIGN KEY (course_id)
        REFERENCES courses (id) ON DELETE SET NULL
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci
  COMMENT='笔记主表';
```

### 3.2 笔记版本表（note_versions）

```sql
CREATE TABLE note_versions (
    id              BIGINT UNSIGNED     NOT NULL AUTO_INCREMENT,
    note_id         BIGINT UNSIGNED     NOT NULL                  COMMENT '笔记ID',
    version_number  INT                 NOT NULL                  COMMENT '版本号',
    title           VARCHAR(500)        NOT NULL                  COMMENT '该版本标题',
    content         LONGTEXT                                      COMMENT '该版本内容',
    created_by      BIGINT UNSIGNED     NOT NULL                  COMMENT '修改者ID',
    created_at      DATETIME(6)         NOT NULL DEFAULT NOW(6),
    PRIMARY KEY (id),
    UNIQUE KEY uk_note_version (note_id, version_number),
    KEY idx_note_id (note_id),
    CONSTRAINT fk_note_versions_note FOREIGN KEY (note_id)
        REFERENCES notes (id) ON DELETE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci
  COMMENT='笔记历史版本表';
```

### 3.3 笔记分享表（note_shares）

```sql
CREATE TABLE note_shares (
    id              BIGINT UNSIGNED     NOT NULL AUTO_INCREMENT,
    note_id         BIGINT UNSIGNED     NOT NULL                  COMMENT '笔记ID',
    shared_by       BIGINT UNSIGNED     NOT NULL                  COMMENT '分享者ID',
    shared_to       BIGINT UNSIGNED     DEFAULT NULL              COMMENT '被分享者ID（NULL=公开分享）',
    share_token     VARCHAR(64)         UNIQUE                    COMMENT '分享链接Token',
    expires_at      DATETIME(6)         DEFAULT NULL              COMMENT '分享过期时间',
    created_at      DATETIME(6)         NOT NULL DEFAULT NOW(6),
    deleted_at      DATETIME(6)         DEFAULT NULL,
    PRIMARY KEY (id),
    KEY idx_note_id (note_id),
    KEY idx_share_token (share_token)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci
  COMMENT='笔记分享记录';
```

## 4. 问答系统表

### 4.1 问题表（questions）

```sql
CREATE TABLE questions (
    id              BIGINT UNSIGNED     NOT NULL AUTO_INCREMENT   COMMENT '问题ID',
    user_id         BIGINT UNSIGNED     NOT NULL                  COMMENT '提问者ID',
    course_id       BIGINT UNSIGNED     DEFAULT NULL              COMMENT '关联课程ID',
    title           VARCHAR(500)        NOT NULL                  COMMENT '问题标题',
    content         TEXT                                          COMMENT '问题详细描述',
    is_resolved     TINYINT(1)          NOT NULL DEFAULT 0        COMMENT '是否已解决',
    view_count      INT                 NOT NULL DEFAULT 0        COMMENT '查看次数',
    answer_count    INT                 NOT NULL DEFAULT 0        COMMENT '回答数量（冗余字段）',
    created_at      DATETIME(6)         NOT NULL DEFAULT NOW(6),
    updated_at      DATETIME(6)         NOT NULL DEFAULT NOW(6)
                    ON UPDATE NOW(6),
    deleted_at      DATETIME(6)         DEFAULT NULL,
    PRIMARY KEY (id),
    KEY idx_user_id (user_id),
    KEY idx_course_id (course_id),
    KEY idx_is_resolved (is_resolved),
    CONSTRAINT fk_questions_user FOREIGN KEY (user_id)
        REFERENCES users (id),
    CONSTRAINT fk_questions_course FOREIGN KEY (course_id)
        REFERENCES courses (id) ON DELETE SET NULL
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci
  COMMENT='课程问答-问题表';
```

### 4.2 回答表（answers）

```sql
CREATE TABLE answers (
    id              BIGINT UNSIGNED     NOT NULL AUTO_INCREMENT   COMMENT '回答ID',
    question_id     BIGINT UNSIGNED     NOT NULL                  COMMENT '关联问题ID',
    user_id         BIGINT UNSIGNED     NOT NULL                  COMMENT '回答者ID',
    content         TEXT                NOT NULL                  COMMENT '回答内容',
    is_accepted     TINYINT(1)          NOT NULL DEFAULT 0        COMMENT '是否被采纳',
    upvote_count    INT                 NOT NULL DEFAULT 0        COMMENT '点赞数',
    created_at      DATETIME(6)         NOT NULL DEFAULT NOW(6),
    updated_at      DATETIME(6)         NOT NULL DEFAULT NOW(6)
                    ON UPDATE NOW(6),
    deleted_at      DATETIME(6)         DEFAULT NULL,
    PRIMARY KEY (id),
    KEY idx_question_id (question_id),
    KEY idx_user_id (user_id),
    CONSTRAINT fk_answers_question FOREIGN KEY (question_id)
        REFERENCES questions (id) ON DELETE CASCADE,
    CONSTRAINT fk_answers_user FOREIGN KEY (user_id)
        REFERENCES users (id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci
  COMMENT='课程问答-回答表';
```

## 9. 审计日志表（audit_logs）

```sql
CREATE TABLE audit_logs (
    id              BIGINT UNSIGNED     NOT NULL AUTO_INCREMENT   COMMENT '日志ID',
    user_id         BIGINT UNSIGNED     DEFAULT NULL              COMMENT '操作用户ID（NULL=系统）',
    action          VARCHAR(200)        NOT NULL                  COMMENT '操作描述',
    resource_type   VARCHAR(100)        DEFAULT NULL              COMMENT '资源类型',
    resource_id     BIGINT UNSIGNED     DEFAULT NULL              COMMENT '资源ID',
    ip_address      VARCHAR(50)         DEFAULT NULL              COMMENT '操作IP',
    user_agent      VARCHAR(500)        DEFAULT NULL              COMMENT '浏览器信息',
    duration        BIGINT              DEFAULT NULL              COMMENT '操作耗时（毫秒）',
    success         TINYINT(1)          NOT NULL DEFAULT 1        COMMENT '是否成功',
    error_message   TEXT                DEFAULT NULL              COMMENT '错误信息',
    created_at      DATETIME(6)         NOT NULL DEFAULT NOW(6)   COMMENT '操作时间',
    PRIMARY KEY (id),
    KEY idx_user_id (user_id),
    KEY idx_action (action),
    KEY idx_created_at (created_at),
    KEY idx_resource (resource_type, resource_id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci
  COMMENT='操作审计日志表';
```
