## 3. 详细步骤（以「课程评价」为例）

### 步骤 1：需求分析

**功能描述**：学生在选课后可以对课程发表评价（评分 + 文字），教师和学生可以查看课程的所有评价。

**API 设计**：
```
POST   /api/courses/{courseId}/reviews     # 发表评价（STUDENT）
GET    /api/courses/{courseId}/reviews     # 查看课程所有评价
DELETE /api/courses/{courseId}/reviews/{reviewId}  # 删除评价（本人或ADMIN）
```

**数据结构**：
- 评价人（student_id）
- 被评价课程（course_id）
- 评分（rating: 1-5）
- 评价内容（content）
- 时间戳

### 步骤 2：创建数据库迁移脚本

**文件路径**：`src/main/resources/db/migration/V16__create_course_reviews_table.sql`

```sql
-- V16__create_course_reviews_table.sql
-- 功能：创建课程评价表
-- 日期：2024-02-01

CREATE TABLE course_reviews (
    id              BIGINT UNSIGNED     NOT NULL AUTO_INCREMENT   COMMENT '评价ID',
    course_id       BIGINT UNSIGNED     NOT NULL                  COMMENT '被评价课程ID',
    student_id      BIGINT UNSIGNED     NOT NULL                  COMMENT '评价者ID',
    rating          TINYINT UNSIGNED    NOT NULL                  COMMENT '评分 1-5',
    content         TEXT                DEFAULT NULL              COMMENT '评价内容',
    created_at      DATETIME(6)         NOT NULL DEFAULT NOW(6)   COMMENT '评价时间',
    updated_at      DATETIME(6)         NOT NULL DEFAULT NOW(6)
                    ON UPDATE NOW(6)                              COMMENT '更新时间',
    deleted_at      DATETIME(6)         DEFAULT NULL              COMMENT '软删除时间',
    PRIMARY KEY (id),
    -- 每个学生每门课只能评价一次
    UNIQUE KEY uk_course_student (course_id, student_id),
    KEY idx_course_id (course_id),
    KEY idx_student_id (student_id),
    CONSTRAINT fk_reviews_course FOREIGN KEY (course_id)
        REFERENCES courses (id) ON DELETE CASCADE,
    CONSTRAINT fk_reviews_student FOREIGN KEY (student_id)
        REFERENCES users (id),
    CONSTRAINT chk_rating CHECK (rating BETWEEN 1 AND 5)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci
  COMMENT='课程评价表';
```

### 步骤 4：创建 Mapper 接口

**文件路径**：`src/main/java/com/example/mapper/CourseReviewMapper.java`

```java
package com.example.coursebuddybackend.mapper;

import com.baomidou.mybatisplus.core.mapper.BaseMapper;
import com.baomidou.mybatisplus.core.metadata.IPage;
import com.baomidou.mybatisplus.extension.plugins.pagination.Page;
import com.example.coursebuddybackend.entity.CourseReview;
import com.example.coursebuddybackend.vo.CourseReviewVO;
import org.apache.ibatis.annotations.Mapper;
import org.apache.ibatis.annotations.Param;
import org.apache.ibatis.annotations.Select;

@Mapper
public interface CourseReviewMapper extends BaseMapper<CourseReview> {

    // 联表查询，带用户信息
    @Select("""
        SELECT
            cr.id,
            cr.course_id,
            cr.student_id,
            u.full_name AS student_name,
            cr.rating,
            cr.content,
            cr.created_at
        FROM course_reviews cr
        LEFT JOIN users u ON cr.student_id = u.id
        WHERE cr.course_id = #{courseId}
            AND cr.deleted_at IS NULL
        ORDER BY cr.created_at DESC
        """)
    IPage<CourseReviewVO> selectReviewsWithStudent(
        Page<CourseReviewVO> page,
        @Param("courseId") Long courseId);

    // 查询课程平均评分
    @Select("""
        SELECT AVG(rating)
        FROM course_reviews
        WHERE course_id = #{courseId}
            AND deleted_at IS NULL
        """)
    Double selectAverageRating(@Param("courseId") Long courseId);
}
```

**实现**：`src/main/java/com/example/service/impl/CourseReviewServiceImpl.java`

```java
package com.example.coursebuddybackend.service.impl;

import com.baomidou.mybatisplus.core.conditions.query.LambdaQueryWrapper;
import com.baomidou.mybatisplus.core.metadata.IPage;
import com.baomidou.mybatisplus.extension.plugins.pagination.Page;
import com.example.coursebuddybackend.dto.CreateCourseReviewDTO;
import com.example.coursebuddybackend.entity.CourseReview;
import com.example.coursebuddybackend.exception.BusinessException;
import com.example.coursebuddybackend.exception.ResourceNotFoundException;
import com.example.coursebuddybackend.mapper.CourseEnrollmentMapper;
import com.example.coursebuddybackend.mapper.CourseReviewMapper;
import com.example.coursebuddybackend.service.CourseReviewService;
import com.example.coursebuddybackend.vo.CourseReviewVO;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.springframework.stereotype.Service;
import org.springframework.transaction.annotation.Transactional;

@Service
@Slf4j
@RequiredArgsConstructor
@Transactional(readOnly = true)
public class CourseReviewServiceImpl implements CourseReviewService {

    private final CourseReviewMapper reviewMapper;
    private final CourseEnrollmentMapper enrollmentMapper;

    @Override
    @Transactional
    public CourseReviewVO createReview(Long courseId, Long studentId,
                                        CreateCourseReviewDTO dto) {
        // 1. 业务校验：学生是否已选课
        boolean enrolled = enrollmentMapper.existsByUserAndCourse(studentId, courseId);
        if (!enrolled) {
            throw new BusinessException("未选课，无法评价");
        }

        // 2. 业务校验：是否已评价过
        Long count = reviewMapper.selectCount(
            new LambdaQueryWrapper<CourseReview>()
                .eq(CourseReview::getCourseId, courseId)
                .eq(CourseReview::getStudentId, studentId));
        if (count > 0) {
            throw new BusinessException("每门课只能评价一次");
        }

        // 3. 创建评价
        CourseReview review = new CourseReview();
        review.setCourseId(courseId);
        review.setStudentId(studentId);
        review.setRating(dto.getRating());
        review.setContent(dto.getContent());
        reviewMapper.insert(review);

        // 4. 构建 VO 返回
        CourseReviewVO vo = new CourseReviewVO();
        vo.setId(review.getId());
        vo.setCourseId(courseId);
        vo.setStudentId(studentId);
        vo.setRating(dto.getRating());
        vo.setContent(dto.getContent());
        vo.setCreatedAt(review.getCreatedAt());
        return vo;
    }

    @Override
    public IPage<CourseReviewVO> listReviews(Long courseId, int page, int size) {
        Page<CourseReviewVO> pageParam = new Page<>(page, size);
        return reviewMapper.selectReviewsWithStudent(pageParam, courseId);
    }

    @Override
    @Transactional
    public void deleteReview(Long courseId, Long reviewId,
                              Long operatorId, String operatorRole) {
        CourseReview review = reviewMapper.selectById(reviewId);
        if (review == null || !review.getCourseId().equals(courseId)) {
            throw new ResourceNotFoundException("评价不存在");
        }

        // 权限校验：本人或管理员才能删除
        boolean isAdmin = "ROLE_ADMIN".equals(operatorRole);
        boolean isOwner = review.getStudentId().equals(operatorId);
        if (!isAdmin && !isOwner) {
            throw new BusinessException("无权删除此评价");
        }

        reviewMapper.deleteById(reviewId);  // 软删除
        log.info("评价 [{}] 已被用户 [{}] 删除", reviewId, operatorId);
    }
}
```

### 步骤 7：编写 Controller

**文件路径**：`src/main/java/com/example/controller/CourseReviewController.java`

```java
package com.example.coursebuddybackend.controller;

import com.baomidou.mybatisplus.core.metadata.IPage;
import com.example.coursebuddybackend.common.ApiResponse;
import com.example.coursebuddybackend.dto.CreateCourseReviewDTO;
import com.example.coursebuddybackend.security.SecurityUtils;
import com.example.coursebuddybackend.service.CourseReviewService;
import com.example.coursebuddybackend.vo.CourseReviewVO;
import io.swagger.v3.oas.annotations.Operation;
import io.swagger.v3.oas.annotations.tags.Tag;
import jakarta.validation.Valid;
import lombok.RequiredArgsConstructor;
import org.springframework.security.access.prepost.PreAuthorize;
import org.springframework.web.bind.annotation.*;

@RestController
@RequestMapping("/courses/{courseId}/reviews")
@RequiredArgsConstructor
@Tag(name = "课程评价", description = "课程评价相关接口")
public class CourseReviewController {

    private final CourseReviewService reviewService;

    @PostMapping
    @PreAuthorize("hasRole('STUDENT')")
    @Operation(summary = "发表课程评价")
    public ApiResponse<CourseReviewVO> createReview(
            @PathVariable Long courseId,
            @Valid @RequestBody CreateCourseReviewDTO dto) {
        Long studentId = SecurityUtils.getCurrentUserId();
        CourseReviewVO vo = reviewService.createReview(courseId, studentId, dto);
        return ApiResponse.success(vo);
    }

    @GetMapping
    @Operation(summary = "查看课程评价列表")
    public ApiResponse<IPage<CourseReviewVO>> listReviews(
            @PathVariable Long courseId,
            @RequestParam(defaultValue = "1") Integer page,
            @RequestParam(defaultValue = "10") Integer size) {
        IPage<CourseReviewVO> result = reviewService.listReviews(courseId, page, size);
        return ApiResponse.success(result);
    }

    @DeleteMapping("/{reviewId}")
    @Operation(summary = "删除评价（本人或管理员）")
    public ApiResponse<Void> deleteReview(
            @PathVariable Long courseId,
            @PathVariable Long reviewId) {
        Long operatorId = SecurityUtils.getCurrentUserId();
        String operatorRole = SecurityUtils.getCurrentUserRole();
        reviewService.deleteReview(courseId, reviewId, operatorId, operatorRole);
        return ApiResponse.success(null);
    }
}
```

### 步骤 8：本地测试

#### 8.1 使用 cURL 测试

```bash
# 1. 登录获取 Token
TOKEN=$(curl -s -X POST http://localhost:8080/api/auth/login \
    -H "Content-Type: application/json" \
    -d '{"username":"student1","password":"password123"}' \
    | python3 -c "import sys,json; print(json.load(sys.stdin)['data']['accessToken'])")

# 2. 发表评价
curl -X POST http://localhost:8080/api/courses/1/reviews \
    -H "Authorization: Bearer $TOKEN" \
    -H "Content-Type: application/json" \
    -d '{"rating": 5, "content": "非常好的课程！"}'

# 3. 查看评价列表
curl http://localhost:8080/api/courses/1/reviews \
    -H "Authorization: Bearer $TOKEN"

# 4. 删除评价
curl -X DELETE http://localhost:8080/api/courses/1/reviews/1 \
    -H "Authorization: Bearer $TOKEN"
```

