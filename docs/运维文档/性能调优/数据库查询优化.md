## 1. MyBatis-Plus 查询优化

### 1.1 全局配置说明

```yaml
# application.yml
mybatis-plus:
  configuration:
    map-underscore-to-camel-case: true        # 下划线转驼峰，避免手动映射
    log-impl: org.apache.ibatis.logging.nologging.NoLoggingImpl  # 生产关闭 SQL 日志
  global-config:
    db-config:
      id-type: auto                           # 主键自增策略
      logic-delete-field: deletedAt          # 逻辑删除字段
      logic-delete-value: "NOW()"            # 删除时间
      logic-not-delete-value: "NULL"         # 未删除标识
```

### 1.2 只查询必要字段

避免 `SELECT *`，使用 `select()` 指定字段：

```java
// ❌ 不推荐：查询全部字段（可能包含大文本 content 字段）
List<Note> notes = noteMapper.selectList(
    new LambdaQueryWrapper<Note>().eq(Note::getUserId, userId)
);

// ✅ 推荐：只查询列表所需字段
List<Note> notes = noteMapper.selectList(
    new LambdaQueryWrapper<Note>()
        .select(Note::getId, Note::getTitle, Note::getCreatedAt, Note::getCourseId)
        .eq(Note::getUserId, userId)
        .orderByDesc(Note::getUpdatedAt)
);
```

### 1.3 使用 Lambda 表达式避免字符串拼写错误

```java
// ❌ 不推荐：字符串字段名容易拼错，重构时不会报错
QueryWrapper<Course> wrapper = new QueryWrapper<>();
wrapper.eq("teacher_id", teacherId).eq("status", "ACTIVE");

// ✅ 推荐：Lambda 类型安全，IDE 重构友好
LambdaQueryWrapper<Course> wrapper = new LambdaQueryWrapper<Course>()
    .eq(Course::getTeacherId, teacherId)
    .eq(Course::getStatus, CourseStatus.ACTIVE);
```

### 1.4 批量操作优化

```java
// ❌ 不推荐：循环单条插入，N 次网络往返
for (Attendance attendance : attendanceList) {
    attendanceMapper.insert(attendance);
}

// ✅ 推荐：批量插入（MyBatis-Plus IService.saveBatch）
// 默认每批 1000 条
attendanceService.saveBatch(attendanceList);

// 或指定批次大小
attendanceService.saveBatch(attendanceList, 500);
```

### 1.5 避免 N+1 查询问题

```java
// ❌ N+1 问题：先查课程列表，再逐一查教师信息
List<Course> courses = courseMapper.selectList(null);
for (Course course : courses) {
    User teacher = userMapper.selectById(course.getTeacherId()); // N 次查询
    course.setTeacher(teacher);
}

// ✅ 推荐：使用 JOIN 或 IN 批量查询
// 方案一：自定义 Mapper XML 使用 JOIN
// CourseMapper.xml
// SELECT c.*, u.username, u.email FROM courses c
// LEFT JOIN users u ON c.teacher_id = u.id WHERE c.deleted_at IS NULL

// 方案二：批量查询后内存关联
List<Course> courses = courseMapper.selectList(wrapper);
Set<Long> teacherIds = courses.stream()
    .map(Course::getTeacherId).collect(Collectors.toSet());
Map<Long, User> teacherMap = userMapper.selectBatchIds(teacherIds)
    .stream().collect(Collectors.toMap(User::getId, Function.identity()));
courses.forEach(c -> c.setTeacher(teacherMap.get(c.getTeacherId())));
```

### 1.6 条件构造器中的 NULL 值处理

```java
// ✅ 使用条件方法，当参数为 null 时自动跳过该条件
public List<Course> searchCourses(String keyword, Long categoryId, String status) {
    return courseMapper.selectList(
        new LambdaQueryWrapper<Course>()
            .like(StringUtils.hasText(keyword), Course::getTitle, keyword)
            .eq(categoryId != null, Course::getCategoryId, categoryId)
            .eq(StringUtils.hasText(status), Course::getStatus, status)
            .orderByDesc(Course::getCreatedAt)
    );
}
```

## 2. 索引策略

### 2.1 核心表索引设计

#### users 表

```sql
-- 主键（自动）
PRIMARY KEY (id)

-- 登录认证（高频查询）
CREATE UNIQUE INDEX idx_users_username ON users (username);
CREATE UNIQUE INDEX idx_users_email ON users (email);

-- 按角色查询（管理员查用户列表）
CREATE INDEX idx_users_role ON users (role, deleted_at);

-- 软删除过滤（几乎所有查询都带此条件）
CREATE INDEX idx_users_deleted_at ON users (deleted_at);
```

#### courses 表

```sql
PRIMARY KEY (id)

-- 教师查询自己的课程（高频）
CREATE INDEX idx_courses_teacher_id ON courses (teacher_id, deleted_at);

-- 课程状态筛选（学生浏览课程列表）
CREATE INDEX idx_courses_status ON courses (status, deleted_at);

-- 课程分类筛选
CREATE INDEX idx_courses_category ON courses (category_id, status, deleted_at);

-- 全文检索课程名称和描述
CREATE FULLTEXT INDEX idx_courses_fulltext ON courses (title, description);
```

#### notes 表

```sql
PRIMARY KEY (id)

-- 用户查询自己的笔记（最高频）
CREATE INDEX idx_notes_user_id ON notes (user_id, deleted_at);

-- 课程维度查询笔记
CREATE INDEX idx_notes_course_id ON notes (course_id, user_id, deleted_at);

-- 笔记更新时间排序（笔记列表默认排序）
CREATE INDEX idx_notes_updated_at ON notes (user_id, updated_at DESC, deleted_at);

-- 乐观锁版本字段（V10 迁移添加）
-- version 字段无需索引
```

#### course_enrollments 表

```sql
PRIMARY KEY (id)

-- 唯一约束：一个学生只能加入一次课程
CREATE UNIQUE INDEX idx_enrollment_unique ON course_enrollments (student_id, course_id, deleted_at);

-- 课程维度：查询课程的所有学生
CREATE INDEX idx_enrollment_course_id ON course_enrollments (course_id, deleted_at);

-- 学生维度：查询学生的所有课程
CREATE INDEX idx_enrollment_student_id ON course_enrollments (student_id, deleted_at);
```

#### assignments 表

```sql
PRIMARY KEY (id)

-- 课程下的作业列表
CREATE INDEX idx_assignments_course_id ON assignments (course_id, deleted_at);

-- 截止时间查询（即将截止的作业）
CREATE INDEX idx_assignments_due_date ON assignments (due_date, deleted_at);
```

#### course_discussions 表

```sql
PRIMARY KEY (id)

-- 课程讨论列表（分页）
CREATE INDEX idx_discussions_course_id ON course_discussions (course_id, created_at DESC, deleted_at);

-- 用户的讨论记录
CREATE INDEX idx_discussions_user_id ON course_discussions (user_id, deleted_at);

-- 楼中楼：回复某条讨论
CREATE INDEX idx_discussions_parent_id ON course_discussions (parent_id, deleted_at);
```

### 2.2 索引使用验证

```sql
-- 使用 EXPLAIN 验证索引使用情况
EXPLAIN SELECT * FROM notes
WHERE user_id = 1 AND deleted_at IS NULL
ORDER BY updated_at DESC
LIMIT 20 OFFSET 0;

-- 关注 type 列：
-- ALL  = 全表扫描（需要优化）
-- ref  = 普通索引扫描（良好）
-- eq_ref = 唯一索引扫描（优秀）
-- const = 主键/唯一键精确匹配（最优）
```

### 2.3 复合索引设计原则

```
遵循最左前缀原则：(a, b, c) 索引可以覆盖 a、a+b、a+b+c 的查询条件

Course Buddy 典型复合索引：
(user_id, deleted_at)        → 用户的有效记录列表
(course_id, user_id, deleted_at) → 课程+用户的有效记录
(status, deleted_at, created_at) → 状态筛选+时间排序
```

## 3. 分页最佳实践（MybatisPlusPageUtils）

### 3.1 MybatisPlusPageUtils 工具类使用

项目封装了 `MybatisPlusPageUtils`，统一分页参数处理：

```java
// 典型用法：课程列表分页
@GetMapping("/courses")
public ApiResponse<IPage<CourseVO>> listCourses(
        @RequestParam(defaultValue = "1") int page,
        @RequestParam(defaultValue = "10") int size,
        @RequestParam(required = false) String keyword) {

    Page<Course> pageParam = MybatisPlusPageUtils.buildPage(page, size);
    // pageParam 已处理：size 上限 100，防止超大分页

    LambdaQueryWrapper<Course> wrapper = new LambdaQueryWrapper<Course>()
        .like(StringUtils.hasText(keyword), Course::getTitle, keyword)
        .eq(Course::getStatus, CourseStatus.ACTIVE)
        .orderByDesc(Course::getCreatedAt);

    IPage<Course> result = courseService.page(pageParam, wrapper);
    return ApiResponse.success(result.convert(courseConverter::toVO));
}
```

### 3.2 深度分页问题及优化

```sql
-- ❌ 深度分页性能问题：OFFSET 100000 需要扫描并丢弃 10 万条记录
SELECT * FROM notes WHERE user_id = 1 AND deleted_at IS NULL
ORDER BY id DESC LIMIT 20 OFFSET 100000;

-- ✅ 游标分页（Keyset Pagination）：基于上一页最后一条记录的 ID
SELECT * FROM notes
WHERE user_id = 1 AND deleted_at IS NULL AND id < :lastId
ORDER BY id DESC LIMIT 20;
```

**前端适配游标分页**：

```java
// 返回下一页游标
public CursorPageResult<NoteVO> listNotesByCursor(Long userId, Long lastId, int size) {
    LambdaQueryWrapper<Note> wrapper = new LambdaQueryWrapper<Note>()
        .eq(Note::getUserId, userId)
        .lt(lastId != null, Note::getId, lastId)
        .isNull(Note::getDeletedAt)
        .orderByDesc(Note::getId)
        .last("LIMIT " + Math.min(size, 100));

    List<Note> notes = noteMapper.selectList(wrapper);
    Long nextCursor = notes.isEmpty() ? null : notes.get(notes.size() - 1).getId();
    return new CursorPageResult<>(notes.stream().map(noteConverter::toVO).toList(),
                                   nextCursor, notes.size() == size);
}
```

### 3.3 COUNT 查询优化

MyBatis-Plus 默认在分页查询时执行 `COUNT(*)` 总量查询，对大表性能影响显著：

```java
// 当不需要总数时，关闭 count 查询
Page<Note> page = new Page<>(pageNum, pageSize, false); // false = 不查总数

// 或者使用近似总数（MySQL 统计信息）
@Select("SELECT TABLE_ROWS FROM information_schema.TABLES " +
        "WHERE TABLE_SCHEMA = DATABASE() AND TABLE_NAME = 'notes'")
Long getApproximateNoteCount();
```

## 4. 慢查询分析与优化

### 4.1 开启 MySQL 慢查询日志

```sql
-- 运行时临时开启（重启后失效）
SET GLOBAL slow_query_log = 'ON';
SET GLOBAL long_query_time = 1;   -- 超过 1 秒记录
SET GLOBAL slow_query_log_file = '/var/lib/mysql/slow.log';
SET GLOBAL log_queries_not_using_indexes = 'ON';  -- 记录未使用索引的查询
```

在 Docker 容器中配置（`docker-compose.yml`）：

```yaml
services:
  mysql:
    command: >
      --character-set-server=utf8mb4
      --collation-server=utf8mb4_unicode_ci
      --slow_query_log=1
      --long_query_time=1
      --slow_query_log_file=/var/lib/mysql/slow.log
      --log_queries_not_using_indexes=1
```

### 4.2 使用 mysqldumpslow 分析慢查询

```bash
# 进入 MySQL 容器
docker exec -it course-buddy-mysql bash

# 统计最慢的 10 条查询
mysqldumpslow -s t -t 10 /var/lib/mysql/slow.log

# 统计出现次数最多的 10 条查询
```

### 4.4 常见慢查询场景与解决方案

#### 场景一：笔记搜索功能（全文检索）

```sql
-- ❌ LIKE '%keyword%' 无法使用索引
SELECT * FROM notes WHERE content LIKE '%Java%' AND user_id = 1;

-- ✅ 使用 MySQL 全文索引
ALTER TABLE notes ADD FULLTEXT INDEX idx_notes_content_ft (title, content);
SELECT * FROM notes
WHERE MATCH(title, content) AGAINST ('Java' IN BOOLEAN MODE)
  AND user_id = 1 AND deleted_at IS NULL;
```

#### 场景二：课程讨论楼中楼（递归查询）

```sql
-- ❌ 递归 CTE 在大数据量时性能差
WITH RECURSIVE discussion_tree AS (...)

-- ✅ 使用 parent_id 限制查询深度，前端分批加载
SELECT * FROM course_discussions
WHERE course_id = ? AND parent_id IS NULL AND deleted_at IS NULL
ORDER BY created_at DESC LIMIT 20;
-- 子评论单独按需加载
SELECT * FROM course_discussions
WHERE parent_id = ? AND deleted_at IS NULL ORDER BY created_at;
```

#### 场景三：考勤统计（聚合查询）

```sql
-- ❌ 每次实时统计所有考勤记录
SELECT student_id, COUNT(*) as total, SUM(is_present) as present_count
FROM attendances WHERE course_id = ? GROUP BY student_id;

-- ✅ 增加复合索引
CREATE INDEX idx_attendances_course_student ON attendances (course_id, student_id, is_present);
-- 或将统计结果缓存到 Redis，定期更新
```

### 4.5 EXPLAIN 输出解读

```sql
EXPLAIN SELECT c.*, u.username FROM courses c
LEFT JOIN users u ON c.teacher_id = u.id
WHERE c.status = 'ACTIVE' AND c.deleted_at IS NULL
ORDER BY c.created_at DESC LIMIT 10;
```

关键字段说明：

| 字段 | 值 | 含义 |
|------|-----|------|
| `type` | `ALL` | 全表扫描，需要优化 |
| `type` | `ref` | 使用了非唯一索引 |
| `type` | `eq_ref` | 使用了唯一索引 |
| `possible_keys` | `idx_courses_status` | 可能使用的索引 |
| `key` | `idx_courses_status` | 实际使用的索引 |
| `rows` | `1000` | 预估扫描行数（越小越好） |
| `Extra` | `Using filesort` | 需要额外排序（影响性能） |
| `Extra` | `Using index` | 覆盖索引（最优） |

## 5. 软删除（deletedAt）查询优化

### 5.1 软删除对查询的影响

MyBatis-Plus 全局逻辑删除配置后，所有 `selectList`、`selectById` 等方法自动追加 `AND deleted_at IS NULL` 条件。

**潜在问题**：`IS NULL` 条件在部分 MySQL 版本中无法充分利用索引。

**解决方案**：

```sql
-- 方案一：复合索引包含 deleted_at，确保 IS NULL 走索引
CREATE INDEX idx_notes_user_deleted ON notes (user_id, deleted_at);

-- 方案二：MySQL 8.0 函数索引（当 deleted_at 值域集中时）
CREATE INDEX idx_notes_active ON notes ((deleted_at IS NULL));

-- 方案三：分区表（超大表时考虑）
-- 将 deleted_at IS NOT NULL 的记录定期归档到历史表
```

### 5.2 查询已删除记录

```java
// 某些场景（如管理员恢复数据）需要查询已删除记录
// 使用 QueryWrapper 的 includeDeleted 方法（MyBatis-Plus 3.5.3+）
noteMapper.selectList(
    new QueryWrapper<Note>()
        .eq("user_id", userId)
        .isNotNull("deleted_at")  // 只查已删除
);
```

## 6. 连接池调优

### 6.1 HikariCP 参数详解

```yaml
spring:
  datasource:
    hikari:
      # 生产环境推荐值（根据并发量调整）
      maximum-pool-size: 20       # 最大连接数
      minimum-idle: 5             # 最小空闲连接
      connection-timeout: 30000   # 获取连接超时（30s）
      idle-timeout: 600000        # 空闲连接回收时间（10min）
      max-lifetime: 1800000       # 连接最大存活时间（30min，需小于 MySQL wait_timeout）
      keepalive-time: 300000      # 连接保活心跳（5min）
      validation-timeout: 5000    # 连接有效性验证超时（5s）
      connection-test-query: SELECT 1  # MySQL 推荐使用此心跳语句
      pool-name: CourseBuddyPool  # 便于监控识别
      leak-detection-threshold: 60000  # 连接泄漏检测（60s 未归还则告警）
```

## 7. 查询缓存策略

### 7.1 哪些查询适合缓存

| 查询类型 | 是否缓存 | 缓存时长 | 说明 |
|---------|---------|---------|------|
| 课程列表（已发布） | ✅ | 10 分钟 | 变化不频繁 |
| 课程详情 | ✅ | 5 分钟 | 热点数据 |
| 用户信息 | ✅ | 30 分钟 | 变化极少 |
| 笔记内容 | ❌ | - | 用户私有，实时性要求高 |
| 考勤统计 | ✅ | 1 小时 | 计算代价高，实时性要求低 |
| 讨论列表 | ❌ | - | 实时性要求高 |

### 7.2 Spring Cache 与 Redis 集成

```java
@Service
public class CourseServiceImpl implements ICourseService {

    @Cacheable(value = "course:detail", key = "#courseId",
               unless = "#result == null")
    public CourseVO getCourseDetail(Long courseId) {
        // 只有缓存未命中时才查询数据库
        return courseMapper.selectById(courseId);
    }

    @CacheEvict(value = "course:detail", key = "#course.id")
    public void updateCourse(Course course) {
        courseMapper.updateById(course);
    }

    @CacheEvict(value = "course:list", allEntries = true)
    public void publishCourse(Long courseId) {
        // 课程发布后清除列表缓存
    }
}
```

详细缓存策略请参考 [Redis 缓存优化文档](./Redis缓存优化.md)。
